## 🔧 **最小限の修正**

**上記の部分を以下に置き換えてください**：

```javascript
// DOMに追加
videoGrid.appendChild(remoteVideo);
console.log(`📺 ${remoteUID}のビデオ要素をDOMに追加`);

// 🔧 修正：安定した再生のため少し待機
setTimeout(() => {
  remoteVideo.play()
    .then(() => {
      console.log(`✅ ${remoteUID}のビデオ再生成功`);
    })
    .catch(err => {
      console.warn(`⚠️ ${remoteUID}のビデオ再生失敗:`, err);
      
      // 手動再生ボタンを追加
      const playBtn = document.createElement("button");
      playBtn.textContent = "▶️ 映像を再生";
      playBtn.style.margin = "10px";
      playBtn.style.padding = "10px";
      playBtn.style.backgroundColor = "#00ff00";
      playBtn.style.border = "none";
      playBtn.style.borderRadius = "5px";
      playBtn.style.cursor = "pointer";
      
      playBtn.onclick = () => {
        remoteVideo.play().then(() => {
          playBtn.remove();
          console.log("✅ 手動再生成功");
        });
      };
      
      videoGrid.appendChild(playBtn);
    });
}, 200); // 200ms待機
```
